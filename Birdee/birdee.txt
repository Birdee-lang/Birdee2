
declare function puts (str as pointer) as int

class genericarray
	private len as int
	private dummy as byte
	public function length() as int
		return len
	end
	public function getRaw() as pointer
		return addressof(dummy)
	end
end

class string
	private buf as byte[]
	private len as int
	public function length() as int
		return len
	end
	public function add(other as string) as string
		len=len+other.len
		return this
	end
	public function getRaw() as pointer
		return buf.getRaw()
	end
end class

function print(str as string)
	puts(str.getRaw())
end

function println(str as string)
	puts(str.getRaw())
	puts("\n".getRaw())
end
